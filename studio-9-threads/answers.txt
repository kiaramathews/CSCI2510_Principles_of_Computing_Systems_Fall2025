6. 
#include <stdio.h>
#include <stdlib.h>
#include <pthread.h>

struct thread_data {
    int arg1;
    char arg2[100];
    int ret;
};

void* thread_entry(void* args) {
    struct thread_data* arg_ptr = (struct thread_data*) args;
    printf("Thread started: arg1 = %d, arg2 = %s\n", arg_ptr->arg1, arg_ptr->arg2);

    arg_ptr->ret = arg_ptr->arg1 * 2;

    return NULL;
}

int main() {
    pthread_t thread;
    struct thread_data data;

    data.arg1 = 21;
    snprintf(data.arg2, sizeof(data.arg2), "Return value example");

    pthread_create(&thread, NULL, thread_entry, &data);
    pthread_join(thread, NULL);

    printf("Returned value from thread: %d\n", data.ret);
    return 0;
}
